  max: 100,
  ttl: 1000 * 60 * 5, // 5 minutes
});

export const cachedApiCall = async <T>(
  key: string,
  apiCall: () => Promise<T>
): Promise<T> => {
  if (cache.has(key)) {
    return cache.get(key) as T;
  }

  const result = await apiCall();
  cache.set(key, result);
  return result;
};
